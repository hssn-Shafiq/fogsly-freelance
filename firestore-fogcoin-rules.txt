// Additional Firestore Security Rules for FOG Coin Settings
// Add these rules to your existing firestore.rules file

// FOG coin settings (admin only)
match /fogCoinSettings/{settingsId} {
  allow read: if request.auth != null;
  allow write: if request.auth != null && 
    exists(/databases/$(database)/documents/admins/$(request.auth.uid));
}

// FOG coin settings history (admin only for write, all authenticated users can read for transparency)
match /fogCoinSettingsHistory/{historyId} {
  allow read: if request.auth != null;
  allow create: if request.auth != null && 
    exists(/databases/$(database)/documents/admins/$(request.auth.uid));
  allow update, delete: if false; // History should be immutable
}
